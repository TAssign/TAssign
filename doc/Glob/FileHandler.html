def read(file)
  if File.exists?(file)
    in_file = File.read(file)
    data = JSON.parse(in_file)
    return data
  else
    return false
  end
end

def write(file, data)
  if File.exists?(file)
    out_file = File.open(file, "w+")
    out_file.write(JSON.pretty_generate(data))
    out_file.close
  else
    File.new(file, "w+")
    write(file, data)
  end
end

def read(file)
  Sysfile.sysfile_status(file)
  if File.exists?(file.name)
    in_file = File.read(file.name)
    data = JSON.parse(in_file)
    return data
  else
    return false
  end
end

def write(file, data)
  Sysfile.sysfile_status(file)
  if File.exists?(file.name)
    out_file = File.open(file.name, "w+")
    out_file.write(JSON.pretty_generate(data))
    out_file.close
  else
    File.new(file.name, "w+")
    write(file, data)
  end
end
